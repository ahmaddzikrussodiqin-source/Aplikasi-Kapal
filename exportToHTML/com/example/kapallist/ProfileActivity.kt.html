279      private fun hitungDurasiBerlayar(perkiraanKeberangkatan: String?): String {
280          if (perkiraanKeberangkatan.isNullOrEmpty()) return "Belum berlayar"
281          try {
282              val parts = perkiraanKeberangkatan.split(" ")
283              if (parts.size == 3) {
284                  val day = parts[0].toInt()
285                  val monthName = parts[1].lowercase()
286                  val year = parts[2].toInt()
287                  val monthNames = arrayOf("januari", "februari", "maret", "april", "mei", "juni", "juli", "agustus", "september", "oktober", "november", "desember")
288                  val month = monthNames.indexOf(monthName)
289                  if (month == -1) return "Tanggal invalid"
290  
291                  val keberangkatanCalendar = Calendar.getInstance().apply {
292                      set(year, month, day, 0, 0, 0)
293                  }
294                  val today = Calendar.getInstance().apply {
295                      set(Calendar.HOUR_OF_DAY, 0)
296                      set(Calendar.MINUTE, 0)
297                      set(Calendar.SECOND, 0)
298                      set(Calendar.MILLISECOND, 0)
299                  }
300  
301                  val diffMillis = today.timeInMillis - keberangkatanCalendar.timeInMillis
302                  val diffDays = (diffMillis / (1000 * 60 * 60 * 24)).toInt()
303                  return if (diffDays >= 0) "$diffDays hari" else "Belum berlayar"
304              }
305          } catch (e: Exception) {
306              Log.e("ProfileActivity", "Error calculating sailing duration: ${e.message}")
307          }
308          return "Belum berlayar"
309      }

311      private fun formatTanggal(tanggal: String): String {
312          if (tanggal.isEmpty()) return ""
313          val parts = tanggal.split("/")
314          if (parts.size == 3) {
315              val day = parts[0]
316              val monthNum = parts[1].toIntOrNull() ?: 1
317              val year = parts[2]
318              val monthNames = arrayOf("Januari", "Februari", "Maret", "April", "Mei", "Juni", "Juli", "Agustus", "September", "Oktober", "November", "Desember")
319              val month = if (monthNum in 1..12) monthNames[monthNum - 1] else "Invalid"
320              return "$day $month $year"
321          }
322          return tanggal
323      }

325      private fun getMonthName(month: Int): String {
326          val months = arrayOf("januari", "februari", "maret", "april", "mei", "juni", "juli", "agustus", "september", "oktober", "november", "desember")
327          return months[month]
