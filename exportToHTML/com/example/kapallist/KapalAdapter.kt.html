1    package com.example.kapallist
2    
3    import android.view.LayoutInflater
4    import android.view.View
5    import android.view.ViewGroup
6    import android.widget.Button
7    import android.widget.TextView
8    import androidx.recyclerview.widget.LinearLayoutManager
9    import androidx.recyclerview.widget.RecyclerView
10   
11   class KapalAdapter(private val kapalList: MutableList<Kapal>, private val activity: DocumentActivity) : RecyclerView.Adapter<KapalAdapter.KapalViewHolder>() {
12   
13       override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): KapalViewHolder {
14           val view = LayoutInflater.from(parent.context).inflate(R.layout.item_kapal, parent, false)
15           return KapalViewHolder(view)
16       }
17   
18       override fun onBindViewHolder(holder: KapalViewHolder, position: Int) {
19           val kapal = kapalList[position]
20           holder.tvNamaKapal.text = kapal.nama ?: ""  // Handle null
21           holder.tvJumlahDokumen.text = "Dokumen: ${kapal.listDokumen.size}"
22   
23           holder.btnEditKapal.setOnClickListener {
24               activity.showEditKapalDialog(position)
25           }
26   
27           holder.btnHapusKapal.setOnClickListener {
28               activity.showHapusKapalDialog(position)
29           }
30   
31           holder.rvDokumen.layoutManager = LinearLayoutManager(holder.itemView.context)
32           holder.rvDokumen.adapter = DokumenAdapter(kapal.listDokumen, activity)
33       }
34   
35       override fun getItemCount(): Int = kapalList.size
36   
37       class KapalViewHolder(itemView: View) : RecyclerView.ViewHolder(itemView) {
38           val tvNamaKapal: TextView = itemView.findViewById(R.id.tv_nama_kapal)
39           val tvJumlahDokumen: TextView = itemView.findViewById(R.id.tv_jumlah_dokumen)
40           val btnEditKapal: Button = itemView.findViewById(R.id.btn_edit_kapal)
41           val btnHapusKapal: Button = itemView.findViewById(R.id.btn_hapus_kapal)
42           val rvDokumen: RecyclerView = itemView.findViewById(R.id.rv_dokumen)
43       }
44   }
