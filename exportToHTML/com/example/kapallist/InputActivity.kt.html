1    package com.example.kapallist
2    
3    import android.app.AlertDialog
4    import android.app.DatePickerDialog
5    import android.content.Intent
6    import android.os.Bundle
7    import android.view.KeyEvent
8    import android.widget.Button
9    import android.widget.EditText
10   import android.widget.LinearLayout
11   import android.widget.TextView
12   import android.widget.Toast
13   import androidx.appcompat.app.AppCompatActivity
14   import androidx.lifecycle.lifecycleScope
15   import com.google.gson.Gson
16   import com.google.gson.reflect.TypeToken
17   import kotlinx.coroutines.launch
18   import java.util.Calendar
19   
20   class InputActivity : AppCompatActivity() {
21       private val listPersiapan = mutableListOf<String>()
22       private val listDokumen = mutableListOf<DokumenKapal>()
23       private lateinit var sharedPref: android.content.SharedPreferences
24       // PERBAIKAN: Tambahkan database Room untuk load nama kapal dari dokumen
25       private lateinit var database: KapalDatabase
26   
27       private var editMode = false
28       private var kapalIndex = -1
29   
30       override fun onCreate(savedInstanceState: Bundle?) {
31           super.onCreate(savedInstanceState)
32           setContentView(R.layout.activity_input)
33   
34           sharedPref = getSharedPreferences("kapal_data", MODE_PRIVATE)
35           // PERBAIKAN: Inisialisasi database Room
36           database = KapalDatabase.getDatabase(this)
37   
38           editMode = intent.getBooleanExtra("edit_mode", false)
39           kapalIndex = intent.getIntExtra("kapal_index", -1)
40           val namaKapalFromIntent = intent.getStringExtra("nama_kapal")
41   
42           val etNamaKapal = findViewById<EditText>(R.id.et_nama_kapal)
43           val etTanggalKembali = findViewById<EditText>(R.id.et_tanggal_kembali)
44           val etPersiapan = findViewById<EditText>(R.id.et_persiapan)
45           val btnTambahPersiapan = findViewById<Button>(R.id.btn_tambah_persiapan)
46           val llPersiapanList = findViewById<LinearLayout>(R.id.ll_persiapan_list)
47           val btnSimpan = findViewById<Button>(R.id.btn_simpan)
48   
49           if (namaKapalFromIntent != null) {
50               etNamaKapal.setText(namaKapalFromIntent)
51           }
52   
53           // PERBAIKAN: Tambahkan onClickListener pada etNamaKapal untuk menampilkan list nama kapal dari Room
54           etNamaKapal.setOnClickListener {
55               lifecycleScope.launch {
56                   val kapalEntities = database.kapalDao().getAllKapal()
57                   val namaKapalList = kapalEntities.map { it.nama }
58                   if (namaKapalList.isNotEmpty()) {
59                       val builder = AlertDialog.Builder(this@InputActivity)
60                       builder.setTitle("Pilih Nama Kapal")
61                       builder.setItems(namaKapalList.toTypedArray()) { _, which ->
62                           etNamaKapal.setText(namaKapalList[which])
63                       }
64                       builder.setNegativeButton("Batal", null)
65                       builder.show()
66                   } else {
67                       Toast.makeText(this@InputActivity, "Tidak ada kapal di dokumen", Toast.LENGTH_SHORT).show()
68                   }
69               }
70           }
71   
72           if (editMode && kapalIndex >= 0) {
73               val gson = Gson()
74               val json = sharedPref.getString("list_kapal", "[]")
75               val type = object : TypeToken<MutableList<Kapal>>() {}.type
76               val kapalList: MutableList<Kapal> = try {
77                   gson.fromJson(json, type) ?: mutableListOf()
78               } catch (e: Exception) {
79                   mutableListOf()
80               }
81               if (kapalIndex < kapalList.size) {
82                   val kapal = kapalList[kapalIndex]
83                   etNamaKapal.setText(kapal.nama)
84                   etTanggalKembali.setText(kapal.tanggalKembali)
85                   listPersiapan.clear()
86                   listPersiapan.addAll(kapal.listPersiapan)
87                   listDokumen.clear()
88                   listDokumen.addAll(kapal.listDokumen)
89                   updatePersiapanListUI(llPersiapanList)
90               }
91           }
92   
93           etPersiapan.setOnKeyListener { _, keyCode, event ->
94               if (event.action == KeyEvent.ACTION_DOWN && keyCode == KeyEvent.KEYCODE_ENTER) {
95                   val persiapanText = etPersiapan.text.toString().trim()
96                   if (persiapanText.isNotEmpty()) {
97                       val items = persiapanText.split("\n").map { it.trim() }.filter { it.isNotBlank() }
98                       if (items.isNotEmpty()) {
99                           listPersiapan.addAll(items)
100                          etPersiapan.text.clear()
101                          updatePersiapanListUI(llPersiapanList)
102                      } else {
103                          Toast.makeText(this, "Persiapan tidak boleh kosong", Toast.LENGTH_SHORT).show()
104                      }
105                  } else {
106                      Toast.makeText(this, "Persiapan tidak boleh kosong", Toast.LENGTH_SHORT).show()
107                  }
108                  true
109              } else {
110                  false
111              }
112          }
113  
114          etTanggalKembali.setOnClickListener {
115              val calendar = Calendar.getInstance()
116              val year = calendar.get(Calendar.YEAR)
117              val month = calendar.get(Calendar.MONTH)
118              val day = calendar.get(Calendar.DAY_OF_MONTH)
119  
120              val datePickerDialog = DatePickerDialog(this, { _, selectedYear, selectedMonth, selectedDay ->
121                  val selectedDate = "$selectedDay/${selectedMonth + 1}/$selectedYear"
122                  etTanggalKembali.setText(selectedDate)
123              }, year, month, day)
124              datePickerDialog.show()
125          }
126  
127          btnTambahPersiapan.setOnClickListener {
128              val persiapanText = etPersiapan.text.toString().trim()
129              if (persiapanText.isNotEmpty()) {
130                  val items = persiapanText.split("\n").map { it.trim() }.filter { it.isNotBlank() }
131                  if (items.isNotEmpty()) {
132                      listPersiapan.addAll(items)
133                      etPersiapan.text.clear()
134                      updatePersiapanListUI(llPersiapanList)
135                  } else {
136                      Toast.makeText(this, "Persiapan tidak boleh kosong", Toast.LENGTH_SHORT).show()
137                  }
138              } else {
139                  Toast.makeText(this, "Persiapan tidak boleh kosong", Toast.LENGTH_SHORT).show()
140              }
141          }
142  
143          btnSimpan.setOnClickListener {
144              val namaKapal = etNamaKapal.text.toString()
145              val tanggalKembali = etTanggalKembali.text.toString()
146  
147              if (namaKapal.isNotEmpty() && tanggalKembali.isNotEmpty() && listPersiapan.isNotEmpty()) {
148                  val kapal = Kapal(
149                      nama = namaKapal,
150                      tanggalKembali = tanggalKembali,
151                      listPersiapan = listPersiapan,
152                      listDokumen = listDokumen
153                  )
154  
155                  val gson = Gson()
156                  val json = sharedPref.getString("list_kapal", "[]")
157                  val type = object : TypeToken<MutableList<Kapal>>() {}.type
158                  val kapalList: MutableList<Kapal> = try {
159                      gson.fromJson(json, type) ?: mutableListOf()
160                  } catch (e: Exception) {
161                      mutableListOf()
162                  }
163  
164                  if (editMode && kapalIndex >= 0 && kapalIndex < kapalList.size) {
165                      kapalList[kapalIndex] = kapal
166                  } else {
167                      kapalList.add(kapal)
168                  }
169  
170                  val editor = sharedPref.edit()
171                  val updatedJson = gson.toJson(kapalList)
172                  editor.putString("list_kapal", updatedJson)
173                  editor.apply()
174  
175                  Toast.makeText(this@InputActivity, "Kapal disimpan!", Toast.LENGTH_SHORT).show()
176                  finish()
177              } else {
178                  Toast.makeText(this, "Nama kapal, tanggal kembali, dan persiapan harus diisi", Toast.LENGTH_SHORT).show()
179              }
180          }
181      }
182  
183      private fun updatePersiapanListUI(llPersiapanList: LinearLayout) {
184          llPersiapanList.removeAllViews()
185          for ((index, persiapan) in listPersiapan.withIndex()) {
186              val tv = TextView(this)
187              tv.text = "${index + 1}. $persiapan"
188              tv.setOnLongClickListener {
189                  listPersiapan.removeAt(index)
190                  updatePersiapanListUI(llPersiapanList)
191                  true
192              }
193              llPersiapanList.addView(tv)
194          }
195      }
196  }
