23   fun hitungTotalHariPersiapan(tanggalInput: Long, tanggalKeberangkatan: Long?): Int? {
24       if (tanggalKeberangkatan == null) return null
25       val selisihMillis = tanggalKeberangkatan - tanggalInput
26       return TimeUnit.MILLISECONDS.toDays(selisihMillis).toInt()
27   }
28   
29   fun hitungDurasiSelesaiPersiapan(tanggalKembali: String?, tanggalSelesai: Long): String? {
30       if (tanggalKembali.isNullOrEmpty()) return null
31       try {
32           val parts = tanggalKembali.split("/")
33           if (parts.size == 3) {
34               val day = parts[0].toInt()
35               val month = parts[1].toInt() - 1
36               val year = parts[2].toInt()
37               val calendar = java.util.Calendar.getInstance()
38               calendar.set(year, month, day, 0, 0, 0)
39               val tanggalKembaliMillis = calendar.timeInMillis
40               val selisihMillis = tanggalSelesai - tanggalKembaliMillis
41               val totalJam = TimeUnit.MILLISECONDS.toHours(selisihMillis).toInt()
42               val hari = totalJam / 24
43               val jam = totalJam % 24
44               return "$hari hari $jam jam"
45           }
46       } catch (e: Exception) {
47           // Handle parsing error
48       }
49       return null
50   }
